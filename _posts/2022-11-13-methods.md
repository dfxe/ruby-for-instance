```rb
# Define a method on a class
class MyClass
  def my_method
    puts "Hello, world!"
  end
end

# Define a method on an instance of a class
class MyClass
  def my_method
    puts "Hello, world!"
  end
end

my_obj = MyClass.new
def my_obj.my_method
  puts "Hello, world!"
end

# Define a method using the self keyword
class MyClass
  def self.my_method
    puts "Hello, world!"
  end
end
```
